``` r
model = function(beta1=0) {
  x = rnorm(30, 0, 1)
  epi = rnorm(30, 0, sd = 50)
  y = 2 + beta1 * x + epi
  data = tibble(x=x,y=y)
  
  ls_fit = lm(y~x, data) %>% 
    broom::tidy()
  return(tibble(beta1 = pull(ls_fit,estimate)[[2]], p = pull(ls_fit,p.value)[2]))
}

n = 100

sim_results = 
  rerun(n, model(0)) %>%  ## rerun
  bind_rows()

sim_all = 
  tibble(beta1 = c(1, 2, 3, 4, 5, 6)) %>% 
  mutate(
    output_list = map(.x = beta1, ~rerun(n, model(beta1 = .x)))) %>% 
  unnest %>% 
  unnest
```

``` r
proportion =
  sim_all %>% 
  filter(p<0.05) %>% 
  group_by(beta1) %>% 
  summarize(proportion = n()/n) 

proportion %>% 
  ggplot(aes(x = beta1, y = proportion)) + geom_point() + geom_line() +
  labs(x = "beta1", y = "Power(proportion of reject)", title = "Association between effect size and power")
```
