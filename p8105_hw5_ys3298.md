#Write a function 
missing_value = function(vec) {
  if (is.numeric(vec)) {
    m = round(mean(vec, na.rm = TRUE), 1)
    output = vec %>% replace_na(m)
  } else if (is.character(vec)) {
    output = vec %>% replace_na("virginica")
  } else {
    stop("Error in type")
  }
  output
}
